import requests
import json

def emotion_detector(text_to_analyze):
    """
    Function to detect emotions using Watson NLP library
    
    Args:
        text_to_analyze (str): Text to be analyzed for emotions
        
    Returns:
        str: Text attribute of the response object from Emotion Detection function
    """
    url = 'https://sn-watson-emotion.labs.skills.network/v1/watson.runtime.nlp.v1/NlpService/EmotionPredict'
    headers = {"grpc-metadata-mm-model-id": "emotion_aggregated-workflow_lang_en_stock"}
    input_json = {"raw_document": {"text": text_to_analyze}}
    
    response = requests.post(url, json=input_json, headers=headers)
    
    # Return the text attribute of the response object
    return response.text
